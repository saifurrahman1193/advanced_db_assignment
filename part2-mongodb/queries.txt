3. A query to list the food items and prices for a specific order.

const order = db.orders.findOne({ "order_id": "order001" });

if (order) {
    const foodItems = order.food_items.map(item => {
        return db.food_items.findOne({ "food_item_id": item.food_item_id }, { "name": 1, "price": 1 });
    });

    foodItems.forEach(item => {
        print(`Food Item: ${item.name}, Price: ${item.price}`);
    });
} else {
    print("Order not found.");
}
